import os
import pandas as pd
import numpy as np
from scipy.stats import norm
import matplotlib.pyplot as plt

def calc_lv(mu, sigma, delta, epsilon, alpha, x, gamma):
    """
    Calculate the liquidity-adjusted lending value.

    Args:
    mu (float): Expected return.
    sigma (float): Volatility of the asset, calculated as the standard deviation of historical asset returns.
    delta (float): The response period given to the client to adjust their position after a margin call.
    epsilon (float): The risk tolerance level, representing the maximum probability of the collateral value falling below the client exposure.
    alpha (float): The threshold level for the margin call.
    x (float): The transaction size.
    gamma (float): The liquidity parameter.

    Returns:
    float: The liquidity-adjusted lending value.
    """
    # calc the inverse CDF of the standard normal distribution for epsilon
    z_epsilon = norm.ppf(epsilon)

    # calc the liquidity-adjusted LV
    num = (1 - alpha) * np.exp(-gamma * x + (mu - sigma**2 / 2) * delta + sigma * np.sqrt(delta) * z_epsilon)
    denom = 1 - alpha * np.exp(-gamma * x + (mu - sigma**2 / 2) * delta + sigma * np.sqrt(delta) * z_epsilon)
    liq_lv = num / denom

    return liq_lv

def calc_bulk_risk(df):
    """
    Calculates the Bulk Risk Indicator for a set of stocks and identifies if the indicator is based on 5xADTV or 3%MC.

    Args:
    df (pandas DataFrame): DataFrame containing the columns 'ADTV', 'Avg MC', 'Avg Close'.

    Returns:
    tuple: Contains two pandas Series.
        1. Bulk Risk Indicator: A Series with the Bulk Risk Indicator for each stock.
        2. Indicator Type: A Series indicating whether the Bulk Risk Indicator is based on '5xADTV' or '3%MC'.
    """
    # calc 5 * ADTV
    five_times_adtv = df['ADTV'] * 5

    # calc the no. of stocks corresponding to 3% of the MC
    three_percent_mc = df['Avg MC'] * 0.03 / df['Avg Close']

    comparison_df = pd.DataFrame({'5xADTV': five_times_adtv, '3%MC': three_percent_mc})

    # calc min
    bulk_risk_indicator = comparison_df.min(axis=1)

    # keep which column (5xADTV or 3%MC)
    indicator_type = comparison_df.idxmin(axis=1)

    return bulk_risk_indicator, indicator_type

def plot_x_lv(ax, df):
    """
    Plots liquidity-adjusted Lending Value (LV) as a function of the number of stocks for each ticker.

    Args:
    ax (matplotlib.axes.Axes): The axes object to draw the plot on.

    Output:
    The function directly plots on the provided axes object. It generates a plot showing LV against the number of stocks,
    including a scatter plot point at the 'Bulk Risk Indicator' value for each ticker.
    """
    delta = 10 / 250  # Response time period in days divided by days in year
    epsilon = 0.01  # Risk tolerance level
    alpha = 0.25  # Margin call threshold

    # generate x values
    x_values = np.logspace(0, 14, 100, base=10)

    # iterate through tickers and plot
    for i, ticker in enumerate(df['Ticker'].unique()):
        y = []
        ticker_df = df[df['Ticker'] == ticker]
        br = ticker_df['Bulk Risk Indicator'].iloc[0]

        for x in x_values:
            mu = (ticker_df['Vola'].iloc[0]**2) / 2
            sigma = ticker_df['Vola'].iloc[0]
            gamma = ticker_df['Mean Gamma Estimate'].iloc[0]

            lv = calc_lv(mu, sigma, delta, epsilon, alpha, x, gamma)
            y.append(lv)
            
        br_lv = calc_lv(mu, sigma, delta, epsilon, alpha, br, gamma)
        ax.scatter(br, br_lv, marker='o')

        linestyle = '-' if i < 10 else '--'
        ax.plot(x_values, y, label=ticker, linestyle=linestyle)

    # axes
    ax.set_xlabel('No. of Stocks (x)')
    ax.set_ylabel('LV')
    ax.set_title('Liquidity-adjusted LV vs Number of Stocks')
    ax.set_xscale('log')  # set x-axis to logarithmic scale
    ax.set_xlim(left=1.1)

    x_ticks = [10**i for i in range(0, 13, 2)]
    x_labels = [f'$10^{i}$' if i < 10 else '$10^{' + str(i) + '}$' for i in range(0, 13, 2)]
    ax.set_xticks(x_ticks, x_labels)

    ax.set_ylim(0, 1)
    ax.set_yticks(np.arange(0, 1.1, 0.1))

    ax.legend(loc='upper right')

def plot_exp_lv(ax, df):
    """
    Plots liquidity-adjusted Lending Value (LV) as a function of the exposure (CHF) for each ticker.

    Args:
    ax (matplotlib.axes.Axes): The axes object to draw the plot on.

    Output:
    The function directly plots on the provided axes object. It generates a plot showing LV against the exposure,
    including a scatter plot point at the 'Bulk Risk Indicator' value for each ticker.
    """
    delta = 10 / 250  # Response time period in days divided by days in year
    epsilon = 0.01  # Risk tolerance level
    alpha = 0.25  # Margin call threshold

    # iterate through tickers and plot
    for i, ticker in enumerate(df['Ticker'].unique()):
        y = []
        ticker_df = df[df['Ticker'] == ticker]
        avg_close = ticker_df['Avg Close'].iloc[0]
        br = ticker_df['Bulk Risk Indicator'].iloc[0] * avg_close

        # generate exposure values (x) as number of stocks times 'Avg Close'
        exposures = np.logspace(0, 12, 100, base=10) * avg_close

        for exposure in exposures:
            num_stocks = exposure / avg_close
            mu = (ticker_df['Vola'].iloc[0]**2) / 2
            sigma = ticker_df['Vola'].iloc[0]
            gamma = ticker_df['Mean Gamma Estimate'].iloc[0]

            lv = calc_lv(mu, sigma, delta, epsilon, alpha, num_stocks, gamma)
            y.append(lv)
            
        br_lv = calc_lv(mu, sigma, delta, epsilon, alpha, br / avg_close, gamma)
        ax.scatter(br, br_lv, marker='o')

        linestyle = '-' if i < 10 else '--'
        ax.plot(exposures, y, label=ticker, linestyle=linestyle)

    # axes
    ax.set_xlabel('Exposure (CHF)')
    ax.set_ylabel('LV')
    ax.set_title('Liquidity-adjusted LV vs Exposure')
    ax.set_xscale('log')  # set x-axis to logarithmic scale
    ax.set_xlim(left=avg_close, right=10**16)  # Start at the smallest average close price

    x_ticks = [10**i for i in range(2, 15, 2)]
    x_labels = [f'$10^{i}$' if i < 10 else '$10^{' + str(i) + '}$' for i in range(2, 15, 2)]
    ax.set_xticks(x_ticks, x_labels)

    ax.set_ylim(0, 1)
    ax.set_yticks(np.arange(0, 1.05, 0.1))

    ax.legend(loc='upper right')
